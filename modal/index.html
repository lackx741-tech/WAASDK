Developer-First

Less Code. More Power.

Copy-paste ready examples for the most common Web3 patterns.

connect.js 
// 3 lines to connect any wallet
import { injectConnectButton } from './sdk/connectButton.js'

injectConnectButton('#app', { theme: 'dark' })
// Done. 400+ wallets supported.
permit2.js 
// Gasless token approval — exact amount
import { signPermitSingle } from './sdk/index.js'

const { signature } = await signPermitSingle(
  provider, account, chainId, {
    token:   '0xTokenAddress',
    amount:  500_000_000n,  // exact — never max
    spender: '0xSpender',
  }
)
multicall.js 
// Batch 3 reads into 1 RPC call
import { multicallRead, buildCall } from './sdk/index.js'

const results = await multicallRead(provider, [
  buildCall('0xContract1', encodedData1),
  buildCall('0xContract2', encodedData2),
  buildCall('0xContract3', encodedData3),
])
eip712.js 
// EIP-712 typed data signing
import { buildDomain, signTypedData } from './sdk/index.js'

const domain = buildDomain({
  name: 'MyApp', version: '1',
  chainId: 1, verifyingContract: '0x…',
})
const sig = await signTypedData(provider, account, typedData)
