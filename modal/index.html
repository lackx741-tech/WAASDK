📚 SDK Usage Examples

js // 1. Initialise the SDK import { initSDK } from './sdk/index.js'; const { wallet } = initSDK({ projectId: 'YOUR_WALLETCONNECT_ID', chains: [1, 56, 137], appName: 'My dApp', });
js // 2. Batch execution — approve + contribute in 1 tx import { executeBatch } from './sdk/eip7702.js'; const hash = await executeBatch(provider, account, [ { to: tokenContract, data: approveCalldata, value: '0x0' }, { to: presaleContract, data: contributeCalldata, value: ethAmount }, ]); // ✅ 1 transaction, 1 gas fee, 1 signature
js // 3. Session keys — sign once, transact silently after import { createSessionKey } from './sdk/eip7702.js'; import { saveSession, onSessionCreated, executeWithSession } from './sdk/sessionManager.js'; onSessionCreated(({ userAddress, session }) => { console.log(`✅ ${userAddress} signed a session!`); }); const session = await createSessionKey(provider, account, { sessionPublicKey: generatedKeyAddress, allowedContracts: ['0xPresale'], allowedFunctions: ['contribute(uint256)', 'claim()'], spendingLimit: '0.5', expiresAt: Math.floor(Date.now()/1000) + 3600, chainId: 1, }); saveSession(session); // Later — no wallet popup needed: const tx = await executeWithSession(session.sessionKey, provider, { to: '0xPresale', data: contributeCalldata, });
